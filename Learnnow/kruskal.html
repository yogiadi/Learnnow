Kruskal

Base step
    Sort the edges of a tree as per their weight.
    Create a set for all the elements with itself as parent and rank as 0.
    Create union of all the sets by their sorted edge and increase their rank.
          
class Vertex:
  def __init__(self,name):
    self.name=name
    self.parent=self
    self.rank=0

def union(dsets1,dsets2):
  link(find_set(dsets1),find_set(dsets2))

def link(dsets1,dsets2):
  if dsets1.rank > dsets2.rank :
    dsets2.parent=dsets1
  else :
    dsets1.parent=dsets2
    if (dsets1.rank == dsets2.rank):
      dsets2.rank=dsets2.rank +1
def find_set(dsets):
  if dsets != dsets.parent:
    dsets.parent=find_set(dsets.parent)
  return dsets.parent

class Graph:
  def __init__(self,name):
    self.name=name
    self.vertices=[]
    self.edges=[]
  def add_vertices(self,vertex):
    if vertex not in self.vertices:
      self.vertices.append(vertex)
  def add_edges(self,weight,point1,point2):
    edge=[weight]+[point1]+[point2]
    if edge not in self.edges:
      self.edges.append(edge)
