Disjoint Sets

Base step
    Each element is traeated as a set .
    Union of these sets contain all the elements of these sets.
    A function find set will return the element representing the set.
    In Union by rank path compression each element has a parent having higher rank hence not all the elements needs to be updated.
          
class  dsets:

  def __init__(self,name):
    self.parent=self
    self.name=name
    self.rank=0

def union(dsets1,dsets2):
  link(find_set(dsets1),find_set(dsets2))

def link(dsets1,dsets2):
  if dsets1.rank > dsets2.rank :
    dsets2.parent=dsets1
  else :
    dsets1.parent=dsets2
    if (dsets1.rank == dsets2.rank):
      dsets2.rank=dsets2.rank +1
def find_set(dsets):
  if dsets != dsets.parent:
    dsets.parent=find_set(dsets.parent)
  return dsets.parent
