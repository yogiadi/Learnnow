Cut Rod

Base step
    For each number n , loop through n by breaking it into (1,n-1),(2,n-2)etc.
    Find maximum sum for each number upto n and store it in array
          
          def cut_rod(p,n):
            print("Calling cut rod function on p and n as ",p , " ", n )
            if n==0:
                print("n is zero so it will return 0")
                return 0
            q=-math.inf
            print("q is ",q)
            for i in range(1,n+1):
                print("value of i is ",i)
                q=max(q,p[i]+cut_rod(p,n-i))
                print("Value of q is as ",q)
            return q

            def memoized_cut_rod (p,n):
                r=[]
                for i in range(0,n+1):
                    r.append(-math.inf)
                print("Value of r is " , r)
                return memoized_cut_rod_aux(p,n,r)
            def memoized_cut_rod_aux(p,n,r):
                if r[n]>=0:
                    print("Value of r[n] is ",r[n])
                    return r[n]
                if n==0:
                    print("n is equal to 0")
                    return 0
                else :
                    q=-math.inf
                    for i in range(1,n+1):
                        print("Value of q before comparison ",q)
                        q=max(q,p[i]+memoized_cut_rod_aux(p,n-i,r))
                        print("Value of q after comparison ",q)
                    r[n]=q
                    print("Value of r[n] is ",r[n])
                    return q
